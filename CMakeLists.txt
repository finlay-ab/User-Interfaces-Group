# CMakeLists.txt
cmake_minimum_required(VERSION 3.16)

project(pollutantmonitor VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets Charts)
qt_standard_project_setup()

qt_add_executable(pollutantmonitor
    main.cpp
    overviewWidget.cpp
)

target_link_libraries(pollutantmonitor PRIVATE 
    Qt6::Widgets
    Qt6::Charts
)

set_target_properties(pollutantmonitor PROPERTIES
    WIN32_EXECUTABLE ON
    MACOSX_BUNDLE OFF
)
